021import java.net.MalformedURLException;
022import java.net.URL;
023import java.util.NoSuchElementException;
024import java.util.Set;
025import java.util.TreeMap;
026import java.util.regex.Matcher;
027import java.util.regex.Pattern;
028import lombok.Data;
029import lombok.Getter;
030import lombok.extern.slf4j.Slf4j;
031import org.apache.commons.text.StringSubstitutor;
032import org.apache.commons.text.lookup.StringLookup;
033import org.apache.maven.artifact.Artifact;
034import org.apache.maven.shared.artifact.filter.StrictPatternIncludesArtifactFilter;
035
036import static java.util.Arrays.asList;
037import static java.util.stream.Collectors.toSet;
038
039/**
040 * {@code <link/>} parameter.
041 *
042 * {@bean.info}
043 *
044 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
045 */
046@Data @Slf4j
047public class Link {
048    private static final Pattern NUMBER = Pattern.compile("[0-9]+");
049    private static final Pattern DOT = Pattern.compile("[.]");
050
051    private String artifact = null;
052    private URL url = null;
053    @Getter(lazy = true)
054    private final StrictPatternIncludesArtifactFilter filter =
055        new StrictPatternIncludesArtifactFilter(asList(artifact.split("[,\\p{Space}]+")));
056
057    /**
058     * See {@link StrictPatternIncludesArtifactFilter#include(Artifact)}.
059     */
060    public boolean include(Artifact artifact) {
061        return getFilter().include(artifact);
062    }
063
064    /**
065     * Method to return a {@link URL} substituting <code>{g}</code>,
066     * <code>{a}</code>, and <code>{v}</code> with
067     * {@link Artifact#getGroupId()}, {@link Artifact#getArtifactId()}, and
068     * {@link Artifact#getVersion()}, respectively.
069     *
070     * @param   artifact        The {@link Artifact}.
071     *
072     * @return  The {@link URL} after substitution.
073     */
074    public URL getUrl(Artifact artifact) {
075        URL url = getUrl();
076
077        if (url != null && artifact != null) {
078            TreeMap<String,String> map = new MapImpl(artifact);
079            StringSubstitutor substitutor = new StringSubstitutor((StringLookup) map, "{", "}", '\\');
080
081            substitutor.setEnableSubstitutionInVariables(true);
082
083            String string = url.toString();
084
085            try {
086                url = new URL(substitutor.replace(string));
087            } catch (MalformedURLException exception) {
088                throw new IllegalArgumentException(exception);
089            } catch (NoSuchElementException exception) {
090                throw new IllegalArgumentException(String.format("No value defined for '%s' in '%s'",
091                                                                 exception.getMessage(), string));
092            }
093        }
094
095        return url;
096    }
097
098    private class MapImpl extends TreeMap<String,String> implements StringLookup {
099        private static final long serialVersionUID = -1285467833760198210L;
100
101        public MapImpl(Artifact artifact) {
102            super(String.CASE_INSENSITIVE_ORDER);
103
104            put("groupId", artifact.getGroupId());
105            put("artifactId", artifact.getArtifactId());
106            put("version", artifact.getBaseVersion());
107
108            put("g", get("groupId"));
109            put("a", get("artifactId"));
110            put("v", get("version"));
111
112            String version = get("version");
113
114            for (String key : asList("major", "minor", "micro", "patch")) {
115                Matcher matcher = NUMBER.matcher(version);
116
117                if (matcher.lookingAt()) {
118                    put(key, matcher.group());
119                    version = version.substring(matcher.group().length());
120                } else {
121                    break;
122                }
123
124                matcher = DOT.matcher(version);
125
126                if (matcher.lookingAt()) {
127                    version = version.substring(matcher.group().length());
128                    continue;
129                } else {
130                    break;
131                }
132            }
133        }
134
135        @Override
136        public String lookup(String key) {
137            String value = get(key);
138
139            if (value == null) {
140                String prefix = key.toLowerCase();
141                Set<String> set =
142                    tailMap(key).entrySet().stream()
143                    .filter(t -> t.getKey().startsWith(prefix))
144                    .map(t -> t.getValue())
145                    .collect(toSet());
146
147                if (set.size() == 1) {
148                    value = set.iterator().next();
149                } else {
150                    throw new NoSuchElementException(key);
151                }
152            }
153
154            return value;
155        }
156    }
157}
