062/**
063 * Abstract base class for javadoc {@link org.apache.maven.plugin.Mojo}s.
064 *
065 * {@bean.info}
066 *
067 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
068 */
069@NoArgsConstructor(access = PROTECTED) @Getter @ToString @Slf4j
070public abstract class AbstractJavadocMojo extends AbstractMojo {
071    @Parameter(defaultValue = "${project.remoteArtifactRepositories}", readonly = true, required = true)
072    private List<ArtifactRepository> remoteRepositories = Collections.emptyList();
073
074    @Parameter(required = false)
075    private Link[] links = new Link[] { };
076
077    @Parameter(required = false)
078    private Offlinelink[] offlinelinks = new Offlinelink[] { };
079
080    @Parameter(defaultValue = "false", property = "javadoc.skip")
081    private boolean skip = false;
082
083    @Inject private MavenSession session = null;
084    @Inject private ArtifactHandlerManager manager = null;
085    @Inject private RepositorySystem system = null;
086    @Inject private ArtifactResolver resolver = null;
087
088    /**
089     * Method to produce a {@link Stream} of
090     * {@link MavenProject#getDependencies()} and
091     * {@link MavenProject#getDependencyManagement()}
092     * {@link Dependency Dependencies}.
093     *
094     * @param   project         The {@link MavenProject}.
095     *
096     * @return  The {@link Dependency} {@link Stream}.
097     */
098    protected Stream<Dependency> getDependencyManagementStream(MavenProject project) {
099        Stream<Dependency> stream =
100            Stream.of(project.getDependencies(), project.getDependencyManagement().getDependencies())
101            .filter(Objects::nonNull)
102            .flatMap(Collection::stream);
103
104        return stream;
105    }
106
107    /**
108     * Method to get the {@link Set} of {@link Link} {@link URL}s.
109     *
110     * @param   project         The {@link MavenProject}.
111     * @param   includeDependencyManagement
112     *                          Whether or not to include dependency
113     *                          management in the analysis.
114     *
115     * @return  The {@link Set} of {@link Link} {@link URL}s.
116     */
117    protected Set<URL> getLinkSet(MavenProject project, boolean includeDependencyManagement) {
118        Set<URL> set = new LinkedHashSet<>();
119
120        for (Link link : links) {
121            project.getArtifacts().stream()
122                .filter(link::include)
123                .map(link::getUrl)
124                .forEach(set::add);
125        }
126
127        if (includeDependencyManagement) {
128            for (Link link : links) {
129                getDependencyManagementStream(project)
130                    .filter(t -> isNotBlank(t.getVersion()))
131                    .map(JavadocArtifact::new)
132                    .filter(link::include)
133                    .map(link::getUrl)
134                    .forEach(set::add);
135            }
136        }
137
138        return set;
139    }
140
141    /**
142     * Method to get the {@link Map} of {@link Offlinelink}
143     * {@link Artifact}s to {@link URL}s.
144     *
145     * @param   project         The {@link MavenProject}.
146     * @param   includeDependencyManagement
147     *                          Whether or not to include dependency
148     *                          management in the analysis.
149     *
150     * @return  The {@link Map} of {@link Offlinelink} {@link Artifact}s to
151     *          {@link URL}s.
152     */
153    protected Map<Artifact,URL> getResolvedOfflinelinkMap(MavenProject project, boolean includeDependencyManagement) {
154        TreeMap<Artifact,URL> map = new TreeMap<>(Comparator.comparing(ArtifactUtils::versionlessKey));
155
156        for (Offlinelink offlinelink : offlinelinks) {
157            project.getArtifacts().stream()
158                .filter(t -> Objects.equals(t.getType(), "jar"))
159                .filter(t -> Objects.equals(t.getClassifier(), "javadoc"))
160                .filter(offlinelink::include)
161                .forEach(t -> map.putIfAbsent(t, offlinelink.getUrl()));
162        }
163
164        Set<Artifact> artifacts =
165            project.getArtifacts().stream()
166            .filter(t -> Objects.equals(t.getType(), "jar"))
167            .filter(t -> isBlank(t.getClassifier()))
168            .map(JavadocArtifact::new)
169            .collect(toCollection(() -> new TreeSet<>(map.comparator())));
170
171        if (includeDependencyManagement) {
172            getDependencyManagementStream(project)
173                .filter(t -> Objects.equals(t.getType(), "jar"))
174                .filter(t -> isBlank(t.getClassifier()))
175                .filter(t -> isNotBlank(t.getVersion()))
176                .map(JavadocArtifact::new)
177                .forEach(artifacts::add);
178        }
179
180        artifacts.removeAll(map.keySet());
181
182        ProjectBuildingRequest request = getProjectBuildingRequest();
183
184        for (Offlinelink offlinelink : offlinelinks) {
185            Set<Artifact> set =
186                artifacts.stream()
187                .filter(offlinelink::include)
188                .collect(toCollection(LinkedHashSet::new));
189
190            for (Artifact artifact : set) {
191                URL url = map.get(artifact);
192
193                if (url == null) {
194                    log.info("Resolving {}...", artifact);
195
196                    try {
197                        map.putIfAbsent(resolver.resolveArtifact(request, artifact).getArtifact(),
198                                        offlinelink.getUrl(artifact));
199                    } catch (Exception exception) {
200                        log.warn("{}: {}", artifact, exception.getMessage());
201                        log.debug("{}", exception);
202                    }
203                } else {
204                    if (! Objects.equals(url, offlinelink.getUrl(artifact))) {
205                        log.warn("{} matches {} but was previously resolved with {}", artifact, offlinelink, url);
206                    }
207                }
208            }
209        }
210
211        return map;
212    }
213
214    private ProjectBuildingRequest getProjectBuildingRequest() {
215        List<ArtifactRepository> repoList = getRemoteRepositories();
216        Settings settings = session.getSettings();
217
218        system.injectMirror(repoList, settings.getMirrors());
219        system.injectProxy(repoList, settings.getProxies());
220        system.injectAuthentication(repoList, settings.getServers());
221
222        ProjectBuildingRequest request = new DefaultProjectBuildingRequest(session.getProjectBuildingRequest());
223
224        request.setRemoteRepositories(repoList);
225
226        return request;
227    }
228
229    @Override
230    public void execute() throws MojoExecutionException, MojoFailureException {
231    }
232
233    private class JavadocArtifact extends DefaultArtifact {
234        public JavadocArtifact(Artifact artifact) {
235            this(artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion());
236        }
237
238        public JavadocArtifact(Dependency dependency) {
239            this(dependency.getGroupId(), dependency.getArtifactId(), dependency.getVersion());
240        }
241
242        public JavadocArtifact(String gav) { this(gav.split("[:]")); }
243
244        public JavadocArtifact(String... gav) {
245            super(gav.length > 0 ? gav[0] : EMPTY,
246                  gav.length > 1 ? gav[1] : EMPTY,
247                  gav.length > 2 ? gav[gav.length - 1] : EMPTY,
248                  EMPTY, "jar", "javadoc", manager.getArtifactHandler("jar"));
249        }
250    }
